\newcommand{\seccao}[1]{
	\chapter*{Secção #1}
	\phantomsection
	\addcontentsline{toc}{chapter}{Secção #1}
}

\newcommand{\exercicio}[1]{
	\section*{#1.}
	\phantomsection
	\addcontentsline{toc}{section}{#1.}
	\setcounter{exercicio}{#1}
	\setcounter{proposition}{0}
}

\newcounter{exercicio}

\newtheorem{proposition}{Proposição}[exercicio]
